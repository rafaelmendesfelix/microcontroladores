float Raiz_Quadrada(float S){
	int n = 0;
	float x = S/2;

	while(n<10){
	x = (x + S/x)/2;
	n++;
	}
	return x;
	}

int Mult(int a, int b){
    int mult=0;
    if(b==0)
    return 0;
    else{
        while(b>0){
        mult = mult + a;
        b--;
        }
    }
return mult;
}

		while(i<N){

int Divisao(int a, int b){
int i=0;
while(a>=b){
	a = a - b;
	i++;
	}
return i;
}

int Resto(int a, int b){
while(a>=b){
    a = a - b;
    }
return a;
}

unsigned long long Fatorial(unsigned long long n){

int i;
i=(n-1);
if(n==0){
    return 1;

    while(i>0){
        n = Mult(n,i);
        i--;
        }
    }

int Palindromo(int *p,int tamanho)
{
	int i;
	for(i=0;i<tamanho;i++)
	{
		if(p[i]!=p[(tamanho-1)-i])
		{
			return 0;
		}
	}
	return 1;
}

{
